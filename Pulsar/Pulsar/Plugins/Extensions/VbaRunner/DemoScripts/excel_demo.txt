Sub PulsarDemo()
    ' Pulsar VBA Demo Script
    ' This script demonstrates basic Excel automation
    Dim ws As Worksheet
    Set ws = ActiveSheet
    
    ' Add a header
    ws.Cells(1, 1).Value = "Hello from Pulsar!"
    ws.Cells(2, 1).Value = "This is an automated VBA demo."
    ws.Cells(3, 1).Value = "Timestamp: " & Now()
    
    ' Format the header
    ws.Range("A1").Font.Bold = True
    ws.Range("A1").Font.Size = 14
    ws.Columns("A").AutoFit
    
    ' Show completion
    MsgBox "VBA demo completed successfully!", vbInformation, "Pulsar"
End Sub
