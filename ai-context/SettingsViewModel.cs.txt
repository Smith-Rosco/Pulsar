using System.Collections.ObjectModel;
using System.Windows;
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using Pulsar.Models;
using Pulsar.Services.Interfaces;
using Pulsar.Views; // 引用 View 层用于 Dialog
using MessageBox = System.Windows.MessageBox;      // 显式指定使用 WPF 的 MessageBox

namespace Pulsar.ViewModels
{
    public partial class SettingsViewModel : ObservableObject
    {
        private readonly IConfigService _configService;
        private AppConfig _config;

        // --- 导航相关 ---
        [ObservableProperty]
        private ObservableCollection<NavigationItem> _navigationItems = new();

        [ObservableProperty]
        private NavigationItem _selectedNavItem;

        // --- 界面状态 ---
        [ObservableProperty] private string _headerTitle = "Loading...";
        [ObservableProperty] private string _headerSubtitle = "";
        [ObservableProperty] private bool _isGlobalSettingsVisible;
        [ObservableProperty] private bool _isSlotsEditorVisible;

        // 当前正在编辑的列表 (绑定到 UI)
        [ObservableProperty]
        private ObservableCollection<GridItem> _currentSlotCollection;

        // 暴露给 UI 绑定的配置对象
        public AppConfig Config => _config;

        // 构造函数注入服务
        public SettingsViewModel(IConfigService configService)
        {
            _configService = configService;
            InitializeAsync();
        }

        private async void InitializeAsync()
        {
            _config = await _configService.LoadAsync();
            OnPropertyChanged(nameof(Config)); // 通知 UI 刷新

            RefreshNavigation();
            SelectedNavItem = NavigationItems.FirstOrDefault(); // 默认选中第一项
        }

        // 当导航项改变时自动触发
        partial void OnSelectedNavItemChanged(NavigationItem value)
        {
            if (value == null) return;

            HeaderTitle = value.Title;
            HeaderSubtitle = value.Subtitle;

            // 切换面板显示
            IsGlobalSettingsVisible = value.Type == NavType.General;
            IsSlotsEditorVisible = value.Type == NavType.Switcher || value.Type == NavType.Profile;

            // 设置当前数据源
            if (value.Type == NavType.Switcher)
            {
                // 包装 List 为 ObservableCollection 以支持 UI 实时增删
                CurrentSlotCollection = new ObservableCollection<GridItem>(_config.Switcher);
                // 监听集合变动并同步回原始 List (简单实现)
                CurrentSlotCollection.CollectionChanged += (s, e) => SyncList(_config.Switcher, CurrentSlotCollection);
            }
            else if (value.Type == NavType.Profile && value.Key != null)
            {
                if (_config.Profiles.TryGetValue(value.Key, out var list))
                {
                    CurrentSlotCollection = new ObservableCollection<GridItem>(list);
                    CurrentSlotCollection.CollectionChanged += (s, e) => SyncList(list, CurrentSlotCollection);
                }
            }
            else
            {
                CurrentSlotCollection = null;
            }
        }

        // --- 命令 (Commands) ---

        [RelayCommand]
        private void AddSlot()
        {
            if (CurrentSlotCollection == null) return;

            // 计算可用 Slot ID
            int nextSlot = 1;
            var usedSlots = CurrentSlotCollection.Select(x => x.Slot).ToHashSet();
            for (int i = 1; i <= 8; i++)
            {
                if (!usedSlots.Contains(i)) { nextSlot = i; break; }
            }

            // 根据当前模式决定类型
            var type = SelectedNavItem.Type == NavType.Switcher
                ? Pulsar.Models.Enums.GridItemType.Launcher
                : Pulsar.Models.Enums.GridItemType.Action;

            CurrentSlotCollection.Add(new GridItem
            {
                Slot = nextSlot,
                Label = "New Item",
                Type = type,
                Process = type == Pulsar.Models.Enums.GridItemType.Launcher ? "process" : null
            });
        }

        [RelayCommand]
        private void DeleteProfile()
        {
            // 修复：使用 MessageBox 别名或全名
            if (SelectedNavItem?.Key != null && _config.Profiles.ContainsKey(SelectedNavItem.Key))
            {
                // 这里原来报错 Ambiguous reference
                if (MessageBox.Show($"Delete profile '{SelectedNavItem.Key}'?", "Confirm", MessageBoxButton.YesNo) == MessageBoxResult.Yes)
                {
                    _config.Profiles.Remove(SelectedNavItem.Key);
                    RefreshNavigation();
                    SelectedNavItem = NavigationItems.FirstOrDefault();
                }
            }
        }

        [RelayCommand]
        private void AddProfile()
        {
            // 简单输入框 (后续可优化为 DialogService)
            var dialog = new SimpleInputDialog("Enter process name (e.g. 'code'):");
            // 注意：SimpleInputDialog 需稍后创建
            if (dialog.ShowDialog() == true && !string.IsNullOrWhiteSpace(dialog.InputText))
            {
                string key = dialog.InputText.ToLower().Replace(".exe", "");
                if (!_config.Profiles.ContainsKey(key))
                {
                    _config.Profiles.Add(key, new List<GridItem>());
                    RefreshNavigation();
                }
            }
        }

        [RelayCommand]
        private async Task SaveAndClose()
        {
            await _configService.SaveAsync(_config);

            // 修复：使用 WpfApplication 或 System.Windows.Application
            WpfApplication.Current.Windows.OfType<Window>().SingleOrDefault(x => x.DataContext == this)?.Close();
        }

        // --- 辅助方法 ---

        private void RefreshNavigation()
        {
            var list = new ObservableCollection<NavigationItem>();

            list.Add(new NavigationItem { Title = "General", Subtitle = "Application behavior.", Icon = "\uE713", Type = NavType.General });
            list.Add(new NavigationItem { Title = "Switcher Mode", Subtitle = "Ctrl+Shift+Q Config.", Icon = "\uE71D", Type = NavType.Switcher });

            foreach (var kvp in _config.Profiles)
            {
                list.Add(new NavigationItem
                {
                    Title = kvp.Key,
                    Subtitle = $"Profile for {kvp.Key}",
                    Icon = "\uE7A7",
                    Type = NavType.Profile,
                    Key = kvp.Key
                });
            }
            NavigationItems = list;
        }

        private void SyncList(List<GridItem> source, ObservableCollection<GridItem> wrapper)
        {
            source.Clear();
            source.AddRange(wrapper);
        }
    }

    // --- 简单数据类 ---
    public enum NavType { General, Switcher, Profile }

    public class NavigationItem
    {
        public string Title { get; set; }
        public string Subtitle { get; set; }
        public string Icon { get; set; }
        public NavType Type { get; set; }
        public string Key { get; set; }
    }
}
