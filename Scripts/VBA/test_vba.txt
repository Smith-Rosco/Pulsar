' @Runner: Test_BasicOperations
' 宏名: Main
' 作用: 演示 MacroFlow 2.0 下的 VBA 基础操作测试（写入、格式化、公式、清除）

Sub Main(Optional ArgFromRunner As Variant)
    On Error GoTo ErrorHandler
    
    ' --- 1. 参数解析层 ---
    Dim targetSheetName As String
    ' 检查是否有来自 MacroFlow 的参数输入，否则弹出输入框
    If Not IsMissing(ArgFromRunner) And CStr(ArgFromRunner) <> "" Then
        targetSheetName = CStr(ArgFromRunner)
    Else
        targetSheetName = InputBox("请输入要进行基础操作测试的工作表名称:", "测试模式", "Sheet1")
        If targetSheetName = "" Then Exit Sub
    End If

    ' --- 2. 业务逻辑层 ---
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Worksheets(targetSheetName)
    
    ' 性能优化：关闭屏幕更新
    Application.ScreenUpdating = False
    
    ' 操作 A: 清除现有内容（初始化）
    ws.Cells.Clear
    
    ' 操作 B: 写入数据标题与模拟数据
    ws.Range("A1").Value = "项目名称"
    ws.Range("B1").Value = "数值"
    ws.Range("A2:A4").Value = Application.Transpose(Array("测试项A", "测试项B", "测试项C"))
    ws.Range("B2").Value = 100
    ws.Range("B3").Value = 200
    ws.Range("B4").Value = 300
    
    ' 操作 C: 设置格式
    With ws.Range("A1:B1")
        .Font.Bold = True                   ' 加粗
        .Interior.Color = RGB(220, 230, 241) ' 背景色
        .HorizontalAlignment = xlCenter      ' 居中
    End With
    
    ' 操作 D: 插入公式
    ws.Range("A5").Value = "总计"
    ws.Range("B5").Formula = "=SUM(B2:B4)"
    ws.Range("B5").Font.Color = vbRed
    
    ' 操作 E: 自动调整列宽
    ws.Columns("A:B").AutoFit

    ' 恢复屏幕更新
    Application.ScreenUpdating = True
    
    MsgBox "工作表 [" & ws.Name & "] 基础操作测试完成！" & vbCrLf & _
           "涵盖：写入、格式化、求和公式。", vbInformation, "MacroFlow Test"

ExitHandler:
    Application.ScreenUpdating = True
    Exit Sub

ErrorHandler:
    MsgBox "测试执行出错: " & Err.Description, vbCritical, "MacroFlow Error"
    Resume ExitHandler
End Sub
